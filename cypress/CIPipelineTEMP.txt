name: CI Pipeline

on: push

jobs:
  setup-project-and-run-cypress:
    name: Pipeline
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Set up Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '20'

      - name: Install dependencies
        run: npm ci

      - name: Run Development Server
        run: |
            npm run dev &
            wait-on http://localhost:5173

      - name: Cypress run
        uses: cypress-io/github-action@v6
        with:
          build: npm run build
          start: npm start

      - name: Stop Development Server
        run: kill $(lsof -t -i:5173)
